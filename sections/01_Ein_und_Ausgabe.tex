	\begin{minipage}[t]{11 cm}		
		\subsection{Streamkonzept}
			\begin{compactitem}
				\item Ein \lc{Stream} repräsentiert einen sequentiellen Datenstrom.
				\item \lc{C++} stellt 4 Standardstreams zur Verfügung:
					\begin{compactitem}
						\item \lc{cin}: Standard-Eingabesteam
						\item \lc{cout}: Standard-Ausgabestream
						\item \lc{cerr}: Standard-Fehlerausgabestream
						\item \lc{clog}: mit \lc{cerr} gekoppelt
					\end{compactitem}
				\item Alle diese Streams können auch mit einer Datei verbunden werden.	
				\item Alle Schlüsselwörter müssen immer ganz links auf der Zeile stehen!\newline
			\end{compactitem} 
	\end{minipage}
